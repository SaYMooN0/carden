```
--апп юзеры
CREATE TABLE app_user ( 
    Id UUID PRIMARY KEY,
    Email VARCHAR(255) UNIQUE NOT NULL,
    PasswordHash VARCHAR(255) NOT NULL,
    RegistrationDate TIMESTAMPTZ NOT NULL
);
--индекс на почту
CREATE INDEX idx_app_users_email ON app_user(Email); 

--неподтвержденные юзеры
CREATE TABLE unconfirmed_user (  
    Id UUID PRIMARY KEY,
    Email VARCHAR(255) UNIQUE NOT NULL,
    PasswordHash VARCHAR(255) NOT NULL,
    ConfirmationCode VARCHAR(128) NOT NULL
);
--индексы для unconfirmed_user
CREATE INDEX idx_unconfirmed_users_email ON unconfirmed_user(Email); 
CREATE INDEX idx_unconfirmed_users_id_code ON unconfirmed_user(Id, ConfirmationCode);

--колоды
CREATE TABLE deck ( 
    Id UUID PRIMARY KEY,
    LastTimeEdited TIMESTAMPTZ NOT NULL
);

--сложный тип для карточек
CREATE TYPE content_item AS ( 
	Value VARCHAR(4096),
	"Type" VARCHAR(30),
	OrderInCard INT
);

--карточки
CREATE TABLE card (  
    Id UUID PRIMARY KEY,
    DeckId UUID NOT NULL,
	
    ContentFront content_item[] NOT NULL,
	ContentBack content_item[] NOT NULL,

    LastTimeEdited TIMESTAMPTZ NOT NULL,
    CreationTime TIMESTAMPTZ NOT NULL,
    FOREIGN KEY (DeckId) REFERENCES deck(Id) ON DELETE CASCADE
);

--растения
CREATE TABLE plant (  
    Id UUID PRIMARY KEY,
    OwnerId UUID NOT NULL,
    Name VARCHAR(100) NOT NULL,
    Description TEXT,
    DeckId UUID NOT NULL UNIQUE,
    PotTypeName VARCHAR(50) NOT NULL,
    PlantSpecieName VARCHAR(50) NOT NULL,
    CreationDate TIMESTAMPTZ NOT NULL,
    FOREIGN KEY (OwnerId) REFERENCES app_user(Id) ON DELETE CASCADE,
    FOREIGN KEY (DeckId) REFERENCES deck(Id) ON DELETE CASCADE
);

--индексы для plant
CREATE INDEX idx_plant_owner_id ON plant(OwnerId);  
CREATE INDEX idx_plant_owner_name ON plant(OwnerId, Name);  
CREATE INDEX idx_plant_owner_creation_date ON plant(OwnerId, CreationDate);

```
