-- app users
CREATE TABLE app_user ( 
   "Id" UUID PRIMARY KEY,
   "Email" VARCHAR(255) UNIQUE NOT NULL,
   "PasswordHash" VARCHAR(255) NOT NULL,
   "RegistrationDate" TIMESTAMPTZ NOT NULL
);

CREATE INDEX idx_app_user_email ON app_user("Email");


-- unconfirmed users
CREATE TABLE unconfirmed_user (  
   "Id" UUID PRIMARY KEY,
   "Email" VARCHAR(255) UNIQUE NOT NULL,
   "PasswordHash" VARCHAR(255) NOT NULL,
   "ConfirmationCode" VARCHAR(128) NOT NULL
);

CREATE INDEX idx_unconfirmed_user_email ON unconfirmed_user("Email");
CREATE INDEX idx_unconfirmed_user_id_confirmation_code
    ON unconfirmed_user("Id", "ConfirmationCode");


-- decks
CREATE TABLE deck ( 
   "Id" UUID PRIMARY KEY,
   "LastTimeEdited" TIMESTAMPTZ NOT NULL
);


-- cards
CREATE TABLE card (  
   "Id" UUID PRIMARY KEY,
   "DeckId" UUID NOT NULL,

   "ContentFront" TEXT[] NOT NULL,
   "ContentBack" TEXT[] NOT NULL,

   "LastTimeEdited" TIMESTAMPTZ NOT NULL,
   "CreationTime" TIMESTAMPTZ NOT NULL,

   -- study data
   "StudyStateType" VARCHAR(50) NULL,
   "StudyDueAt" TIMESTAMPTZ NULL,
   "StudyLearningStepIndex" INTEGER NULL,
   "StudyStartedAt" TIMESTAMPTZ NULL,
   "StudyReviewInterval" INTERVAL NULL,
   "StudyLastReviewedAt" TIMESTAMPTZ NULL,

   CONSTRAINT fk_card_deck
      FOREIGN KEY ("DeckId") REFERENCES deck("Id") ON DELETE CASCADE
);

CREATE INDEX idx_card_deck_id ON card("DeckId");
CREATE INDEX idx_card_deck_creation_time ON card("DeckId", "CreationTime");
CREATE INDEX idx_card_deck_last_time_edited ON card("DeckId", "LastTimeEdited");

-- for review cards loading
CREATE INDEX idx_card_deck_study_due_at
    ON card("DeckId", "StudyDueAt", "CreationTime", "Id")
    WHERE "StudyStateType" IS NOT NULL AND "StudyDueAt" IS NOT NULL;

-- for new cards loading
CREATE INDEX idx_card_deck_new_cards
    ON card("DeckId", "CreationTime", "Id")
    WHERE "StudyStateType" IS NULL;


-- plants
CREATE TABLE plant (  
   "Id" UUID PRIMARY KEY,
   "OwnerId" UUID NOT NULL,
   "Name" VARCHAR(100) NOT NULL,
   "DeckId" UUID NOT NULL UNIQUE,
   "PotTypeName" VARCHAR(50) NOT NULL,
   "PlantSpecieName" VARCHAR(50) NOT NULL,
   "CreationDate" TIMESTAMPTZ NOT NULL,
   "CompletedStudySessionsCount" INTEGER NOT NULL DEFAULT 0,

   CONSTRAINT chk_plant_completed_study_sessions_count
      CHECK ("CompletedStudySessionsCount" >= 0),

   CONSTRAINT fk_plant_owner
      FOREIGN KEY ("OwnerId") REFERENCES app_user("Id") ON DELETE CASCADE,

   CONSTRAINT fk_plant_deck
      FOREIGN KEY ("DeckId") REFERENCES deck("Id") ON DELETE CASCADE
);

CREATE INDEX idx_plant_owner_id ON plant("OwnerId");
CREATE INDEX idx_plant_owner_name ON plant("OwnerId", "Name");
CREATE INDEX idx_plant_owner_creation_date ON plant("OwnerId", "CreationDate");
